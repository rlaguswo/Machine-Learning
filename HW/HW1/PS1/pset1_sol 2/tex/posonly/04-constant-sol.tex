\begin{answer}

	We have
	%
	\begin{align*}
		p(y^{(i)} = 1\mid x^{(i)})
		& = p(y^{(i)} = 1\land t^{(i)} = 1\mid x^{(i)}) \quad\text{(from previous sub-question)}\\
		& = p(t^{(i)} = 1\mid x^{(i)})p(y^{(i)} = 1\mid t^{(i)} = 1, x^{(i)})\\
		& = p(t^{(i)} = 1\mid x^{(i)}) \alpha .
	\end{align*}
	%
	Dividing both sides by $\alpha$, we have the desired result.
\end{answer}
